```testGemini.py
import google.generativeai as genai

# Gemini APIキー設定
genai.configure(api_key="XXXXXX")

model = genai.GenerativeModel("gemini-1.5-pro")

with open("sample2.wav", "rb") as f:
    audio_data = f.read()

response = model.generate_content([
    "この音声を文字起こししてください：話者を分け、00:00形式でタイムスタンプを付けてください。",
    {
        "mime_type": "audio/wav",
        "data": audio_data
    }
])

print(response.text)
```
